    **Step 1: Update DotMDService.index()** in `backend/src/dotmd/api/service.py`:

    ```python
    def index(self, directory: Path) -> IndexStats:
        """Index all markdown files under *directory*.

        Delegates entirely to :class:`IndexingPipeline`.
        ...
        """
        return self._pipeline.index(directory)
    ```

    To:
    ```python
    def index(self, directory: Path, *, force: bool = False) -> IndexStats:
        """Index all markdown files under *directory*.

        Delegates entirely to :class:`IndexingPipeline`.

        Parameters
        ----------
        directory:
            Root directory to scan.
        force:
            When ``True``, bypass incremental change detection and
            re-index all files from scratch.  When ``False`` (default),
            only new and modified files are processed.

        Returns
        -------
        IndexStats
            Summary statistics for the completed index.
        """
        return self._pipeline.index(directory, force=force)
    ```

    **Step 2: Create test file** `backend/tests/test_service_force.py`:

    Use `unittest.mock.patch` to mock `IndexingPipeline.index` and verify the `force` parameter is threaded through:

    ```python
    """Tests for force parameter threading through DotMDService."""
    from unittest.mock import patch, MagicMock
    from pathlib import Path

    import pytest

    # We need to mock the heavy imports before DotMDService tries to load them.
    # Use a simpler approach: patch pipeline.index on an existing service instance.


    class TestServiceForceParameter:
        """Verify DotMDService.index passes force to pipeline."""

        def test_force_false_by_default(self, tmp_path: Path):
            """service.index(dir) should pass force=False to pipeline."""
            with patch("dotmd.api.service.IndexingPipeline") as MockPipeline:
                mock_instance = MagicMock()
                mock_instance.index.return_value = MagicMock()
                MockPipeline.return_value = mock_instance
                # Also mock properties accessed in __init__
                mock_instance.vector_store = MagicMock()
                mock_instance.graph_store = MagicMock()
                mock_instance.metadata_store = MagicMock()

                from dotmd.api.service import DotMDService
                from dotmd.core.config import Settings

                settings = Settings(index_dir=tmp_path / "idx")
                service = DotMDService(settings=settings)
                service.index(tmp_path)

                mock_instance.index.assert_called_once_with(tmp_path, force=False)

        def test_force_true_passed_through(self, tmp_path: Path):
            """service.index(dir, force=True) should pass force=True to pipeline."""
            with patch("dotmd.api.service.IndexingPipeline") as MockPipeline:
                mock_instance = MagicMock()
                mock_instance.index.return_value = MagicMock()
                MockPipeline.return_value = mock_instance
                mock_instance.vector_store = MagicMock()
                mock_instance.graph_store = MagicMock()
                mock_instance.metadata_store = MagicMock()

                from dotmd.api.service import DotMDService
                from dotmd.core.config import Settings

                settings = Settings(index_dir=tmp_path / "idx")
                service = DotMDService(settings=settings)
                service.index(tmp_path, force=True)

                mock_instance.index.assert_called_once_with(tmp_path, force=True)
    ```

    Note: The exact mocking approach may need adjustment depending on how DotMDService.__init__ accesses pipeline properties. The executor should read service.py __init__ to determine which pipeline attributes need mocking (vector_store, graph_store, metadata_store are accessed to create search engines). If the mock approach is too complex due to __init__ side effects, an alternative is to instantiate DotMDService normally and then replace `service._pipeline.index` with a MagicMock after init.
