# Plan 02 — Pipeline Refactor: embed_tracker → meta_tracker, e_text storage, metadata-only fast path

## Objective

Refactor IndexingPipeline to implement the dual-encoder architecture:

1. Replace `embed_tracker` (body+title+tags) with `meta_tracker` (title+tags only)
2. Refactor `_embed_chunks()` to hash raw chunk.text, store `e_text` in VecComponentStore
3. Add `_embed_meta_and_fuse()` for the full encode+fuse path (new file + metadata changes)
4. Rewrite the metadata-only fast path in `index_file()` to read stored e_text BLOBs,
   call TEI once for e_meta, compute e_fused locally, update vec0
5. Add schema version wipe-on-deploy to `_ensure_tables()`
6. Remove `enrich_with_title_and_tags` from the embedding path in content_handlers.py
7. Remove `embed_checksum` from reader.py (Plan 01 already added meta_checksum as replacement)

This is the largest change in the phase. It touches pipeline.py significantly but follows
patterns already established in the codebase (FileTracker, VecComponentStore, struct.pack).

```python
def _embed_chunks(
    self, chunks: list[Chunk],
) -> tuple[list[list[float]], dict[str, str]]:
    """Embed chunk body text (e_text) and store components.

    text_hash is now hash(chunk.text only) — no title/tags prefix.
    e_text vectors are stored in vec_components for metadata-only recompute path.

    Returns (e_text_vectors, text_hashes).
    e_text_vectors are NOT fused — caller must call _compute_and_store_fused().
    """
    if not chunks:
        return [], {}

    # text_hash on raw chunk.text (no enrichment) — true cross-strategy reuse
    text_hashes: dict[str, str] = {
        c.chunk_id: blake3.blake3(c.text.encode()).hexdigest()
        for c in chunks
    }

    # Cache lookup: vec_meta text_hash lookup
    existing: dict[str, list[float]] = {}
    if hasattr(self._vector_store, "lookup_embeddings_by_text_hash"):
        # NOTE: vec_meta still stores text_hash, but now text_hash = hash(body only).
        # On first deploy after schema wipe, this lookup returns empty (correct).
        existing = self._vector_store.lookup_embeddings_by_text_hash(
            list(text_hashes.values()),
        )

    # Global embedding cache lookup
    global_hits = self._embedding_cache.lookup(list(text_hashes.values()))
    for k, v in global_hits.items():
        existing.setdefault(k, v)

    # Also check vec_components for already-stored e_text (avoids re-encode on
    # metadata-only change when e_text is already stored)
    component_hits = self._vec_components.get_batch(
        [c.chunk_id for c in chunks], "text"
    )

    hits = 0
    to_encode_indices: list[int] = []
    embeddings: list[list[float] | None] = [None] * len(chunks)

    for i, chunk in enumerate(chunks):
        th = text_hashes[chunk.chunk_id]
        if th in existing:
            embeddings[i] = existing[th]
            hits += 1
        elif chunk.chunk_id in component_hits:
            embeddings[i] = component_hits[chunk.chunk_id]
            hits += 1
        else:
            to_encode_indices.append(i)

    if to_encode_indices:
        texts_to_encode = [chunks[i].text for i in to_encode_indices]
        new_embeddings = self._semantic_engine.encode_batch(texts_to_encode)
        for j, idx in enumerate(to_encode_indices):
            embeddings[idx] = new_embeddings[j]

        for j, idx in enumerate(to_encode_indices):
            th = text_hashes[chunks[idx].chunk_id]
            self._embedding_cache.store(th, new_embeddings[j])

    # Store e_text in vec_components for all chunks (idempotent for cached)
    for i, chunk in enumerate(chunks):
        if embeddings[i] is not None:
            self._vec_components.store(chunk.chunk_id, "text", embeddings[i])

    total = len(chunks)
    logger.info(
        "embed_text: %d chunks, %d hits (%.1f%%), %d computed",
        total, hits, hits / total * 100 if total else 0, len(to_encode_indices),
    )

    return embeddings, text_hashes  # type: ignore[return-value]
```

```python
@staticmethod
def _normalize_vector(v: list[float]) -> list[float]:
    """Normalize a float vector to unit length. Returns v unchanged if magnitude is 0."""
    import math
    mag = math.sqrt(sum(x * x for x in v))
    return [x / mag for x in v] if mag > 0.0 else list(v)

def _fuse_vectors(
    self,
    e_text: list[float],
    e_meta: list[float],
    weights: dict[str, float],
) -> list[float]:
    """Compute e_fused = normalize(w_text * norm(e_text) + w_meta * norm(e_meta)).

    Uses stdlib math only (no numpy). Pure local computation — no TEI calls.
    """
    w_text = weights.get("text", 0.7)
    w_meta = weights.get("meta", 0.3)
    nt = self._normalize_vector(e_text)
    nm = self._normalize_vector(e_meta)
    raw = [w_text * a + w_meta * b for a, b in zip(nt, nm)]
    return self._normalize_vector(raw)

def _embed_meta_component(self, file_info: FileInfo) -> list[float]:
    """Encode title+tags for a file into e_meta, store in vec_components.

    One TEI call per file. The resulting vector is stored in vec_components
    keyed by (str(file_info.path), 'meta') for reuse on body-only changes.
    """
    title = file_info.frontmatter.get("title", "") if file_info.frontmatter else ""
    tags = file_info.frontmatter.get("tags", []) if file_info.frontmatter else []
    tags_str = ", ".join(str(t) for t in tags) if tags else ""
    meta_text = f"{title} {tags_str}".strip() if (title or tags_str) else title
    result = self._semantic_engine.encode_batch([meta_text])
    e_meta = result[0]
    path_str = str(file_info.path)
    self._vec_components.store(path_str, "meta", e_meta)
    return e_meta

def _compute_and_store_fused(
    self,
    chunks: list[Chunk],
    e_text_vectors: list[list[float]],
    e_meta: list[float],
) -> list[list[float]]:
    """Compute e_fused for each chunk and return the list of fused vectors.

    e_fused = normalize(w_text * norm(e_text_i) + w_meta * norm(e_meta))
    where e_meta is shared across all chunks of the same file.

    The returned e_fused_vectors are what gets stored in vec0.
    """
    weights = self._settings.parsed_embedding_weights
    e_fused_vectors = [
        self._fuse_vectors(e_text, e_meta, weights)
        for e_text in e_text_vectors
    ]
    return e_fused_vectors
```

These methods are added to IndexingPipeline class body. They are pure computation
helpers — no side effects on vec0 directly; the callers handle vec0 writes.

```python
def _check_schema_version(self) -> None:
    """Check schema version sentinel; wipe all vector state if version mismatch.

    Schema version is stored in vec_config_<strategy>_<model> as key='schema_version'.
    Version '2' = Phase 999.12 dual-encoder schema (text_hash on body only).

    On version mismatch or missing sentinel:
    - Clear embedding_cache (text_hash semantics changed)
    - Clear vec_components (stale e_text BLOBs)
    - Delete vec0 and vec_meta tables (stale enriched vectors)
    - Clear chunk_tracker and meta_tracker fingerprints
    - Write new sentinel

    """

    # Read stored schema version
    row = conn.execute(
        f"SELECT value FROM {self._vector_store._CONFIG_TABLE} WHERE key = 'schema_version'"
    ).fetchone()
    stored_version = row[0] if row else None

    if stored_version == self.SCHEMA_VERSION:
        return  # Up to date

    logger.warning(
        "schema_version mismatch: stored=%s expected=%s — wiping all vector state. "
        "trickle will rebuild from scratch. Expected rebuild time: several days.",
        stored_version, self.SCHEMA_VERSION,
    )

    # Wipe embedding cache
```
